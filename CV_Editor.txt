    <script>
        function fetchKeywords() {
            const input = document.getElementById('jobInput').value;
            if (!input) {
                alert("Please enter a job title first!");
                return;
            }

            // UI State: Loading
            document.getElementById('results').classList.add('hidden');
            document.getElementById('loading').classList.remove('hidden');
            
            // Simulate API Network Delay (1.5 seconds)
            setTimeout(() => {
                document.getElementById('loading').classList.add('hidden');
                document.getElementById('results').classList.remove('hidden');
                document.getElementById('displayJob').innerText = input;
                
                // Mock logic to return different keywords based on input text
                const keywords = getMockKeywords(input.toLowerCase());
                const badgeContainer = document.getElementById('keywordBadges');
                
                badgeContainer.innerHTML = keywords.map(kw => 
                    `<span class="bg-blue-100 text-blue-800 text-sm font-semibold px-3 py-1 rounded-full">${kw}</span>`
                ).join('');

            }, 1500);
        }

        // Simple mock function to return relevant keywords
        function getMockKeywords(jobTitle) {
            if (jobTitle.includes('market') || jobTitle.includes('seo')) {
                return ['SEO/SEM', 'Content Strategy', 'Google Analytics', 'Campaign Management', 'Copywriting'];
            } else if (jobTitle.includes('software') || jobTitle.includes('develop') || jobTitle.includes('engineer')) {
                return ['Agile Methodologies', 'Git Version Control', 'RESTful APIs', 'Problem Solving', 'System Design'];
            } else if (jobTitle.includes('design') || jobTitle.includes('ui') || jobTitle.includes('ux')) {
                return ['Figma', 'User Research', 'Wireframing', 'Prototyping', 'Adobe Creative Suite'];
            } else {
                return ['Project Management', 'Cross-functional Collaboration', 'Data Analysis', 'Client Communication', 'Problem Solving'];
            }
        }
    </script>
